## Operator logikal NOT

<details>
  <summary>Contoh output.</summary>

  public static void main (String[] args) {

         boolean lulus;
         
         int markah=83;

         if (markah >=40)

                 lulus=true;

         else

                 lulus=false;
         
         if (!lulus)
                
                System.out.println("Anda perlu mengulangi ujian " );
         
         else
                
                System.out.println("Anda lulus " );
  }
